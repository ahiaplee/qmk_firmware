python3 -m pip install --user qmk
qmk setup

// Compilation databse
qmk generate-compilation-database -kb keychron/q3/ansi -km default

// Non Via
qmk compile -kb keychron/q3/ansi -km default
or
make keychron/q3/ansi:default

// With Via
qmk compile -kb keychron/q3/ansi -km via
or
make keychron/q3/ansi:via

// If via folder missing
cp keyboards/keychron/q3/ansi/keymaps/default/keymap.c keyboards/keychron/q3/ansi/keymaps/via/keymap.c
echo VIA_ENABLE = yes > keyboards/keychron/q3/ansi/keymaps/via/rules.mk
